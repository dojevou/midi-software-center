```python
def detect_from_filename(filename):
    tags = []
    name_lower = filename.lower()

    # Song structure
    structures = {
        "intro": "intro",
        "outro": "outro",
        "verse": "verse",
        "chorus": "chorus",
        "cha": "chorus-a",
        "chb": "chorus-b",
        "ch3": "chorus-3",
        "pre-chorus": "pre-chorus",
        "prechorus": "pre-chorus",
        "bridge": "bridge",
        "middle8": "middle-8",
        "mid8": "middle-8",
        "bkdn": "breakdown",
        "breakdown": "breakdown",
        "ta": "turnaround",
    }

    for key, tag in structures.items():
        if key in name_lower:
            tags.append(tag)

    # BPM detection (enhanced)
    import re
    bpm_match = re.search(r'(\d{2,3})[\s_]?bpm', name_lower)
    if bpm_match:
        bpm = int(bpm_match.group(1))
        tags.append(f"{bpm}-bpm")

        # Genre hints from BPM
        if 136 <= bpm <= 145:
            tags.append("dubstep")
        elif 172 <= bpm <= 176:
            tags.append("dnb")
        elif 120 <= bpm <= 130:
            tags.append("house-tempo")
        elif bpm == 140:
            tags.append("psy-trance")

    # World instruments
    world_instruments = {
        "djembe": ["djembe", "africa", "percussion"],
        "tabla": ["tabla", "asia", "percussion"],
        "darabuka": ["darabuka", "middle-east", "percussion"],
        "kendang": ["kendang", "asia", "percussion"],
        "samul": ["samul-nori", "korea", "percussion"],
        "rebana": ["rebana", "asia", "percussion"],
        "ghatam": ["ghatam", "india", "percussion"],
        "dhol": ["dhol", "india", "percussion"],
        "riq": ["riq", "middle-east", "percussion"],
        "duff": ["duff", "middle-east", "percussion"],
    }

    for key, instr_tags in world_instruments.items():
        if key in name_lower:
            tags.extend(instr_tags)

    # Production elements
    if "loop" in name_lower:
        tags.append("loop")
    if "_seq" in name_lower or " seq" in name_lower:
        tags.append("sequence")
    if "sub" in name_lower and "pad" in name_lower:
        tags.extend(["sub-pad", "pad"])
    elif "pad" in name_lower:
        tags.append("pad")
    if "pluck" in name_lower:
        tags.append("plucked")

    # Mood/style descriptors (Psy Trance)
    moods = {
        "zap": "energetic",
        "wobbly": "wobble",
        "whistle": "bright",
        "waterfalling": "flowing",
        "dark": "dark",
        "intense": "intense",
    }

    for key, mood in moods.items():
        if key in name_lower:
            tags.append(mood)

    return tags
```
