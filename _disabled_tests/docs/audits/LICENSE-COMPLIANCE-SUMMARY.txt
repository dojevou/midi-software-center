VERIFICATION COMMANDS

# Verify no GPL licenses (should return 0)
cargo metadata --format-version 1 | \
  jq '.packages[] | select(.license | contains("GPL"))' | wc -l

# Verify no AGPL licenses (should return 0)
cargo metadata --format-version 1 | \
  jq '.packages[] | select(.license | contains("AGPL"))' | wc -l

# List all unique licenses
cargo metadata --format-version 1 | \
  jq '.packages[] | select(.license != null) | .license' | \
  sort | uniq -c | sort -rn

# Count packages by license type
cargo metadata --format-version 1 | \
  jq '[.packages[] | .license] | group_by(.) | map({license: .[0], count: length})' | \
  jq 'sort_by(.count) | reverse[]'
